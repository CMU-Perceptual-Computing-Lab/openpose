1161  template<typename TDatums, typename TWorker, typename TQueue>
1162  void Wrapper<TDatums, TWorker, TQueue>::configureThreadManager()
1163  {
1164  try
1165  {
1166  // The less number of queues -> the less lag
1167 
1168  // Security checks
1169  if (!mConfigured)
1170  error("Configure the Wrapper class before calling `start()`.", __LINE__, __FUNCTION__, __FILE__);
1171  if ((wDatumProducer == nullptr) == (mUserInputWs.empty())
1172  && mThreadManagerMode != ThreadManagerMode::Asynchronous
1173  && mThreadManagerMode != ThreadManagerMode::AsynchronousIn)
1174  {
1175  const auto message = "You need to have 1 and only 1 producer selected. You can introduce your own"
1176  " producer by using setWorkerInput() or use the OpenPose default producer by"
1177  " configuring it in the configure function) or use the"
1178  " ThreadManagerMode::Asynchronous(In) mode.";
1179  error(message, __LINE__, __FUNCTION__, __FILE__);
1180  }
1181  if (mOutputWs.empty() && mUserOutputWs.empty() && spWGui == nullptr
1182  && mThreadManagerMode != ThreadManagerMode::Asynchronous
1183  && mThreadManagerMode != ThreadManagerMode::AsynchronousOut)
1184  {
1185  error("No output selected.", __LINE__, __FUNCTION__, __FILE__);
1186  }
1187 
1188  // Thread Manager:
1189  // Clean previous thread manager (avoid configure to crash the program if used more than once)
1190  mThreadManager.reset();
1191  mThreadId = 0ull;
1192  auto queueIn = 0ull;
1193  auto queueOut = 1ull;
1194  // If custom user Worker and uses its own thread
1195  spWIdGenerator = std::make_shared<WIdGenerator<std::shared_ptr<TDatums>>>();
1196  if (!mUserInputWs.empty() && mUserInputWsOnNewThread)
1197  {
1198  // Thread 0, queues 0 -> 1
1199  mThreadManager.add(mThreadId, mUserInputWs, queueIn++, queueOut++);
1200  if (spWScaleAndSizeExtractor != nullptr && spWCvMatToOpInput != nullptr)
1201  {
1202  threadIdPP();
1203  // Thread 1, queues 1 -> 2
1204  if (spWCvMatToOpOutput == nullptr)
1205  mThreadManager.add(mThreadId, {spWIdGenerator, spWScaleAndSizeExtractor, spWCvMatToOpInput},
1206  queueIn++, queueOut++);
1207  else
1208  mThreadManager.add(mThreadId, {spWIdGenerator, spWScaleAndSizeExtractor, spWCvMatToOpInput,
1209  spWCvMatToOpOutput}, queueIn++, queueOut++);
1210  }
1211  else
1212  mThreadManager.add(mThreadId, spWIdGenerator, queueIn++, queueOut++);
1213  }
1214  // If custom user Worker in same thread or producer on same thread
1215  else
1216  {
1217  std::vector<TWorker> workersAux;
1218  // Custom user Worker
1219  if (!mUserInputWs.empty())
1220  workersAux = mergeVectors(workersAux, mUserInputWs);
1221  // OpenPose producer
1222  else if (wDatumProducer != nullptr)
1223  workersAux = mergeVectors(workersAux, {wDatumProducer});
1224  // Otherwise
1225  else if (mThreadManagerMode != ThreadManagerMode::Asynchronous
1226  && mThreadManagerMode != ThreadManagerMode::AsynchronousIn)
1227  error("No input selected.", __LINE__, __FUNCTION__, __FILE__);
1228  // ID generator
1229  workersAux = mergeVectors(workersAux, {spWIdGenerator});
1230  // Scale & cv::Mat to OP format
1231  if (spWScaleAndSizeExtractor != nullptr && spWCvMatToOpInput != nullptr)
1232  workersAux = mergeVectors(workersAux, {spWScaleAndSizeExtractor,
1233  spWCvMatToOpInput});
1234  // cv::Mat to output format
1235  if (spWCvMatToOpOutput != nullptr)
1236  workersAux = mergeVectors(workersAux, {spWCvMatToOpOutput});
1237  // Thread 0 or 1, queues 0 -> 1
1238  mThreadManager.add(mThreadId, workersAux, queueIn++, queueOut++);
1239  }
1240  threadIdPP();
1241  // Pose estimation & rendering
1242  // Thread 1 or 2...X, queues 1 -> 2, X = 2 + #GPUs
1243  if (!spWPoses.empty())
1244  {
1245  if (mMultiThreadEnabled)
1246  {
1247  for (auto& wPose : spWPoses)
1248  {
1249  mThreadManager.add(mThreadId, wPose, queueIn, queueOut);
1250  threadIdPP();
1251  }
1252  }
1253  else
1254  {
1255  if (spWPoses.size() > 1)
1256  log("Multi-threading disabled, only 1 thread running. All GPUs have been disabled but the"
1257  " first one, which is defined by gpuNumberStart (e.g. in the OpenPose demo, it is set"
1258  " with the `--num_gpu_start` flag).", Priority::High);
1259  mThreadManager.add(mThreadId, spWPoses.at(0), queueIn, queueOut);
1260  }
1261  queueIn++;
1262  queueOut++;
1263  }
1264  // If custom user Worker and uses its own thread
1265  if (!mUserPostProcessingWs.empty() && mUserPostProcessingWsOnNewThread)
1266  {
1267  // Post processing workers
1268  if (!mPostProcessingWs.empty())
1269  {
1270  // Thread 2 or 3, queues 2 -> 3
1271  mThreadManager.add(mThreadId, mPostProcessingWs, queueIn++, queueOut++);
1272  threadIdPP();
1273  }
1274  // User processing workers
1275  // Thread 3 or 4, queues 3 -> 4
1276  mThreadManager.add(mThreadId, mUserPostProcessingWs, queueIn++, queueOut++);
1277  threadIdPP();
1278  // Output workers
1279  if (!mOutputWs.empty())
1280  {
